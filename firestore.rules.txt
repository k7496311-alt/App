rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Specific matches first
    match /users/{userId} {
      allow read, write: if request.auth != null;
    }
    
    match /plans/{planId} {
      allow read, write: if true; 
    }
    
    match /investments/{invId} {
      allow read, write: if request.auth != null;
    }
    
    match /transactions/{txId} {
      allow read, write: if request.auth != null;
    }
    
    match /system/{docId} {
      allow read, write: if request.auth != null;
    }

    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
